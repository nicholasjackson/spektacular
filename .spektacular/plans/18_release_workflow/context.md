func (s *SpektacularCI) Build(
    ctx context.Context,
    src *dagger.Directory,
    version string,
) (*dagger.Directory, error) {
    if s.hasError() {
        return nil, s.lastError
    }

    // Initialize cache volume if not set
    if s.goCacheVolume == nil {
        s.goCacheVolume = dag.CacheVolume("spektacular-go-cache")
    }

    golang := dag.Container().
        From("golang:1.25").
        WithMountedDirectory("/src", src).
        WithWorkdir("/src").
        WithMountedCache("/go/pkg/mod", dag.CacheVolume("go-mod")).
        WithMountedCache("/root/.cache/go-build", s.goCacheVolume)

    output := dag.Directory()
    
    oses := []string{"darwin", "linux"}
    arches := []string{"amd64", "arm64"}
    
    for _, goos := range oses {
        for _, goarch := range arches {
            build := golang.
                WithEnvVariable("GOOS", goos).
                WithEnvVariable("GOARCH", goarch).
                WithEnvVariable("CGO_ENABLED", "0").
                WithExec([]string{
                    "go", "build",
                    "-ldflags", fmt.Sprintf("-X github.com/jumppad-labs/spektacular/cmd.version=%s", version),
                    "-o", fmt.Sprintf("build/%s/%s/spektacular", goos, goarch),
                    ".",
                })
            
            output = output.WithDirectory(
                fmt.Sprintf("%s/%s", goos, goarch),
                build.Directory(fmt.Sprintf("build/%s/%s", goos, goarch)),
            )
        }
    }
    
    return output, nil
}
```

func (s *SpektacularCI) Archive(
    ctx context.Context,
    builds *dagger.Directory,
    version string,
) (*dagger.Directory, error) {
    if s.hasError() {
        return nil, s.lastError
    }

    archiver := dag.Container().From("alpine:latest").
        WithExec([]string{"apk", "add", "tar", "gzip"})

    output := dag.Directory()
    checksums := ""

    oses := []string{"darwin", "linux"}
    arches := []string{"amd64", "arm64"}

    for _, goos := range oses {
        for _, goarch := range arches {
            archiveName := fmt.Sprintf("spektacular_%s_%s_%s.tar.gz", version, goos, goarch)
            
            archive := archiver.
                WithMountedDirectory("/build", builds.Directory(fmt.Sprintf("%s/%s", goos, goarch))).
                WithWorkdir("/build").
                WithExec([]string{"tar", "czf", fmt.Sprintf("/%s", archiveName), "spektacular"}).
                File(fmt.Sprintf("/%s", archiveName))
            
            output = output.WithFile(archiveName, archive)
            
            // Calculate checksum
            checksum, _ := archiver.
                WithFile(fmt.Sprintf("/%s", archiveName), archive).
                WithExec([]string{"sha256sum", fmt.Sprintf("/%s", archiveName)}).
                Stdout(ctx)
            checksums += checksum
        }
    }

    // Write checksums file
    output = output.WithNewFile("checksums.txt", checksums)
    
    return output, nil
}
```

**Complexity**: Medium
**Token estimate**: ~8k
**Agent strategy**: Single agent, sequential execution

**Key adaptations from jumppad**:
- Archive naming: `spektacular_{version}_{os}_{arch}.tar.gz`
- No Windows archives (removed from OS list)

func (s *SpektacularCI) getVersion(ctx context.Context) (string, error) {
    if s.hasError() {
        return "", s.lastError
    }

    // Use Dagger GitHub module to read PR labels
    gh := dag.Github()
    pr, err := gh.GetPullRequest(ctx, "jumppad-labs", "spektacular", os.Getenv("GITHUB_REF"))
    if err != nil {
        return "0.0.0", nil // No PR found
    }

    labels := pr.Labels()
    
    // Check for release labels
    for _, label := range labels {
        switch label {
        case "release/patch":
            return bumpPatch(currentVersion), nil
        case "release/minor":
            return bumpMinor(currentVersion), nil
        case "release/major":
            return bumpMajor(currentVersion), nil
        }
    }

    return "0.0.0", nil
}

func (s *SpektacularCI) GithubRelease(
    ctx context.Context,
    archives *dagger.Directory,
    version string,
    githubToken *dagger.Secret,
) error {
    if s.hasError() {
        return s.lastError
    }

    // Use Dagger GitHub module to create release
    gh := dag.Github().WithToken(githubToken)
    
    release, err := gh.CreateRelease(ctx, "jumppad-labs", "spektacular", version, dagger.CreateReleaseOpts{
        Name: fmt.Sprintf("spektacular %s", version),
        Body: fmt.Sprintf("Release %s", version),
    })
    if err != nil {
        s.lastError = err
        return err
    }

    // Upload all archives as assets
    entries, _ := archives.Entries(ctx)
    for _, entry := range entries {
        if strings.HasSuffix(entry, ".tar.gz") {
            file := archives.File(entry)
            release.UploadAsset(ctx, entry, file)
        }
    }

    return nil
}
```

func (s *SpektacularCI) UpdateBrew(
    ctx context.Context,
    archives *dagger.Directory,
    version string,
    githubToken *dagger.Secret,
) error {
    if s.hasError() {
        return s.lastError
    }

    // Generate formula content
    darwinAmd64SHA := getArchiveSHA(archives, fmt.Sprintf("spektacular_%s_darwin_amd64.tar.gz", version))
    darwinArm64SHA := getArchiveSHA(archives, fmt.Sprintf("spektacular_%s_darwin_arm64.tar.gz", version))

    formula := fmt.Sprintf(`
class Spektacular < Formula
  desc "AI-powered planning and implementation workflow tool"
  homepage "https://github.com/jumppad-labs/spektacular"
  version "%s"

  on_macos do
    if Hardware::CPU.arm?
      url "https://github.com/jumppad-labs/spektacular/releases/download/%s/spektacular_%s_darwin_arm64.tar.gz"
      sha256 "%s"
    else
      url "https://github.com/jumppad-labs/spektacular/releases/download/%s/spektacular_%s_darwin_amd64.tar.gz"
      sha256 "%s"
    end
  end

  def install
    bin.install "spektacular"
  end
end
`, version, version, version, darwinArm64SHA, version, version, darwinAmd64SHA)

    // Clone homebrew-repo, update formula, commit, push
    git := dag.Container().From("alpine/git:latest").
        WithSecretVariable("GITHUB_TOKEN", githubToken).
        WithExec([]string{"git", "clone", "https://github.com/jumppad-labs/homebrew-repo.git"}).
        WithWorkdir("/homebrew-repo").
        WithNewFile("Formula/spektacular.rb", formula).
        WithExec([]string{"git", "add", "Formula/spektacular.rb"}).
        WithExec([]string{"git", "commit", "-m", fmt.Sprintf("Update spektacular to %s", version)}).
        WithExec([]string{"git", "push"})

    _, err := git.Sync(ctx)
    if err != nil {
        s.lastError = err
        return err
    }

    return nil
}

### Phase 3.5: Release Orchestration

func (s *SpektacularCI) Release(
    ctx context.Context,
    archives *dagger.Directory,
    githubToken *dagger.Secret,
    gemfuryToken *dagger.Secret,
) (string, error) {
    if s.hasError() {
        return "", s.lastError
    }

    // Get version from PR labels
    version, err := s.getVersion(ctx)
    if err != nil {
        s.lastError = err
        return "", err
    }

    if version == "0.0.0" {
        s.lastError = fmt.Errorf("no version to release, did you tag the PR?")
        return "", s.lastError
    }

    // Publish to all channels
    if err := s.GithubRelease(ctx, archives, version, githubToken); err != nil {
        return "", err
    }

    if err := s.UpdateBrew(ctx, archives, version, githubToken); err != nil {
        return "", err
    }

    if err := s.UpdateGemFury(ctx, version, gemfuryToken, archives); err != nil {
        return "", err
    }

    return version, nil
}
```

**Complexity**: Low
**Token estimate**: ~5k
**Agent strategy**: Single agent, sequential execution

        with:

          VERSION=$(dagger call release \
            --archives=./archives \
            --github-token=env:GH_TOKEN \
            --gemfury-token=env:FURY_TOKEN)

          GH_TOKEN: ${{ secrets.GH_TOKEN }}
